import requests

def emotion_detector(text_to_analyze):
    # This URL is specifically for the Skills Network lab environment to avoid connection errors
    url = 'https://sn-watson-emotion.labs.skills.network/v1/watson.runtime.nlp.v1/NlpService/EmotionPredict'
    
    # Required headers for the Watson NLP service
    header = {"grpc-metadata-mm-model-id": "emotion_aggregated-workflow_lang_en_stock"}
    
    # The data object containing the text to be analyzed
    myobj = { "raw_document": { "text": text_to_analyze } }
    
    # Sending the POST request to the emotion detection service
    response = requests.post(url, json = myobj, headers=header)
    
    # Returning the response text (the raw data)
    return response.text
